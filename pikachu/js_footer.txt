(function() {
  // Expose window.__hf for screenshot capture mode (REQUIRED by hyperframes)
  window.__hf = {
    duration: 23,
    seek: function(time) {
      // Seeking is handled by re-rendering at specific frames
      // For now, just return the requested time
      return time;
    }
  };

  // Seeded PRNG (mulberry32) — deterministic across renders
  function mulberry32(seed) {
    return function() {
      seed |= 0; seed = seed + 0x6D2B79F5 | 0;
      var t = Math.imul(seed ^ seed >>> 15, 1 | seed);
      t = t + Math.imul(t ^ t >>> 7, 61 | t) ^ t;
      return ((t ^ t >>> 14) >>> 0) / 4294967296;
    };
  }
  var random = mulberry32(42);

  // Generate floating particles (deterministic)
  var particlesEl = document.getElementById('particles');
  for (var i = 0; i < 25; i++) {
    var p = document.createElement('div');
    p.className = 'particle';
    p.style.left = (random() * 100) + '%';
    p.style.animationDuration = (6 + random() * 8) + 's';
    p.style.animationDelay = (random() * 8) + 's';
    var size = 2 + random() * 3;
    p.style.width = size + 'px';
    p.style.height = size + 'px';
    particlesEl.appendChild(p);
  }

  // Wait for GSAP to be ready, then drive the timeline
  var comp = document.getElementById('comp');
  var scene1 = document.getElementById('scene1');
  var scene2 = document.getElementById('scene2');
  var scene3 = document.getElementById('scene3');
  var scene4 = document.getElementById('scene4');

  var s1Duration = 5;    // seconds
  var s2Duration = 7;
  var s3Duration = 5;
  var s4Duration = 6;
  var totalDuration = s1Duration + s2Duration + s3Duration + s4Duration; // 23s

  var startTime = null;
  var progressFill = document.getElementById('progressFill');

  var s2ProgressStart = s1Duration;
  var s2ProgressEnd = s1Duration + s2Duration;

  function getElapsed() {
    if (!startTime) return 0;
    // Use performance.now() for higher precision
    return (performance.now() - startTime) / 1000;
  }

  function tick() {
    var elapsed = getElapsed();
    var progress = Math.min(elapsed / totalDuration, 1);

    // Progress bar
    if (elapsed >= s1Duration && elapsed <= s2ProgressEnd) {
      var s2p = (elapsed - s1Duration) / s2Duration;
      progressFill.style.width = (s2p * 100) + '%';
    }

    // Scene 1: Trigger at specific times (in seconds)
    if (elapsed > 0.5) {
      document.getElementById('pikachuReveal').classList.add('reveal-anim');
    }
    if (elapsed > 2.0) {
      document.getElementById('yellowFlash').classList.add('flash-anim');
    }
    if (elapsed > 3.0) {
      document.getElementById('brandStamp').classList.add('stamp-anim');
    }

    // Scene 1 → 2 transition
    if (elapsed >= s1Duration && elapsed < s1Duration + 0.1) {
      scene1.classList.remove('active');
      scene2.classList.add('active');
      setTimeout(function() {
        document.getElementById('ec1').classList.add('card-anim');
        document.getElementById('ec2').classList.add('card-anim');
        document.getElementById('ec3').classList.add('card-anim');
        document.getElementById('ec4').classList.add('card-anim');
      }, 50);
    }

    // Scene 2 → 3 transition
    if (elapsed >= s1Duration + s2Duration && elapsed < s1Duration + s2Duration + 0.1) {
      scene2.classList.remove('active');
      scene3.classList.add('active');
      progressFill.style.width = '0%';
      setTimeout(function() {
        document.getElementById('centerProduct').classList.add('prod-anim');
        document.getElementById('heartfelt').classList.add('heart-anim');
      }, 50);
    }

    // Scene 3 → 4 transition
    if (elapsed >= s1Duration + s2Duration + s3Duration && elapsed < s1Duration + s2Duration + s3Duration + 0.1) {
      scene3.classList.remove('active');
      scene4.classList.add('active');
      document.getElementById('closingGlow').classList.add('glow-anim');
      setTimeout(function() {
        document.getElementById('ctaProduct').classList.add('cta-anim');
        document.getElementById('ctaMain').classList.add('cta-text-anim');
        document.getElementById('priceBadge').classList.add('price-anim');
        document.getElementById('finalStamp').classList.add('stamp-final-anim');
      }, 50);
    }

    if (elapsed < totalDuration) {
      requestAnimationFrame(tick);
    } else {
      // Loop — reset everything
      scene4.classList.remove('active');
      scene1.classList.add('active');
      document.getElementById('pikachuReveal').classList.remove('reveal-anim');
      document.getElementById('yellowFlash').classList.remove('flash-anim');
      document.getElementById('brandStamp').classList.remove('stamp-anim');
      document.getElementById('ec1').classList.remove('card-anim');
      document.getElementById('ec2').classList.remove('card-anim');
      document.getElementById('ec3').classList.remove('card-anim');
      document.getElementById('ec4').classList.remove('card-anim');
      document.getElementById('centerProduct').classList.remove('prod-anim');
      document.getElementById('heartfelt').classList.remove('heart-anim');
      document.getElementById('ctaProduct').classList.remove('cta-anim');
      document.getElementById('ctaMain').classList.remove('cta-text-anim');
      document.getElementById('priceBadge').classList.remove('price-anim');
      document.getElementById('finalStamp').classList.remove('stamp-final-anim');
      document.getElementById('closingGlow').classList.remove('glow-anim');
      progressFill.style.width = '0%';
      startTime = performance.now();
      tick();
    }
  }

  // Start when page is ready
  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', function() {
      startTime = performance.now();
      tick();
    });
  } else {
    startTime = performance.now();
    tick();
  }
})();
</script>
</body>
</html>
